```python
# -*- coding: utf-8 -*-
"""
教材：《内河航道与通航水力学》
章节：航道渐变流与通航安全耦合仿真
功能：基于 Manning 公式与渐变流方程计算沿程水深，进一步评估船舶富余水深（UKC）、
输出 KPI 结果表，并绘制水力与通航安全图。
章节关键词：渐变流、Manning 糙率、弗劳德数、富余水深、通航能力
"""

import numpy as np
from scipy.integrate import solve_ivp
from scipy.optimize import brentq
import matplotlib.pyplot as plt

# =========================
# 1) 关键参数（可按教学场景调整）
# =========================
g = 9.81                  # 重力加速度 (m/s^2)
L = 50_000.0              # 航段长度 (m)
b = 80.0                  # 航道等效底宽 (m)
S0 = 1.2e-4               # 河床坡降 (-)
n = 0.028                 # Manning 糙率
Q = 900.0                 # 设计流量 (m^3/s)
y_upstream = 7.0          # 上游控制断面水深 (m)

ship_draft = 3.2          # 船舶吃水 (m)
squat_coeff = 0.15        # 下沉量系数（经验）
ukc_required = 0.8        # 最小富余水深要求 (m)

nx = 400                  # 沿程离散点数

# =========================
# 2) 水力学基础函数
# =========================
def area(y):
    """矩形断面过水面积 A"""
    return b * y

def wetted_perimeter(y):
    """矩形断面湿周 P"""
    return b + 2.0 * y

def hydraulic_radius(y):
    """水力半径 R = A/P"""
    return area(y) / wetted_perimeter(y)

def velocity(y):
    """断面平均流速 V = Q/A"""
    return Q / area(y)

def friction_slope(y):
    """摩阻坡降 Sf（Manning）"""
    A = area(y)
    R = hydraulic_radius(y)
    return (n * Q / (A * R ** (2.0 / 3.0))) ** 2

def froude_number(y):
    """弗劳德数 Fr = V/sqrt(g*y)（矩形断面）"""
    V = velocity(y)
    return V / np.sqrt(g * y)

# =========================
# 3) 求正常水深 yn（用于对比）
# =========================
def manning_residual(y):
    A = area(y)
    R = hydraulic_radius(y)
    q_calc = (1.0 / n) * A * R ** (2.0 / 3.0) * np.sqrt(S0)
    return q_calc - Q

yn = brentq(manning_residual, 0.2, 30.0)

# =========================
# 4) 渐变流微分方程 dy/dx = (S0 - Sf)/(1 - Fr^2)
# =========================
def gvf_ode(x, y):
    yv = max(float(y[0]), 0.05)  # 防止数值迭代出现非物理负水深
    sf = friction_slope(yv)
    fr = froude_number(yv)

    den = 1.0 - fr ** 2
    # 避免分母过小导致数值发散
    if abs(den) < 1e-4:
        den = 1e-4 if den >= 0 else -1e-4

    dydx = (S0 - sf) / den
    return [dydx]

x_eval = np.linspace(0.0, L, nx)
sol = solve_ivp(
    gvf_ode, (0.0, L), [y_upstream],
    t_eval=x_eval, method="RK45",
    rtol=1e-6, atol=1e-8
)

if not sol.success:
    raise RuntimeError(f"数值积分失败：{sol.message}")

x = sol.t
y = np.maximum(sol.y[0], 0.05)

# 河床与水面线高程（以 x=0 断面河床高程为 0）
z_bed = -S0 * x
z_ws = z_bed + y

# =========================
# 5) 通航指标（UKC）计算
# =========================
V = velocity(y)
Fr = froude_number(y)
Sf = friction_slope(y)

# 船舶下沉量（简化经验模型）
squat = squat_coeff * V ** 2 / (2.0 * g)

# 富余水深 UKC = 实际水深 - 吃水 - 下沉量
ukc = y - ship_draft - squat
safe_mask = ukc >= ukc_required
safe_ratio = 100.0 * np.mean(safe_mask)

i_min_y = int(np.argmin(y))
i_min_ukc = int(np.argmin(ukc))

# =========================
# 6) KPI 结果表打印
# =========================
kpi_rows = [
    ("正常水深 yn (m)", yn),
    ("最小水深 y_min (m)", y[i_min_y]),
    ("最小水深位置 x (km)", x[i_min_y] / 1000.0),
    ("最大流速 V_max (m/s)", np.max(V)),
    ("最大弗劳德数 Fr_max (-)", np.max(Fr)),
    ("平均摩阻坡降 Sf_mean (-)", np.mean(Sf)),
    ("最小富余水深 UKC_min (m)", ukc[i_min_ukc]),
    ("UKC 满足率 (%)", safe_ratio),
]

def print_kpi_table(rows):
    name_w = max(len(r[0]) for r in rows) + 2
    val_w = 14
    print("\n=== KPI 结果表 ===")
    print(f"{'指标'.ljust(name_w)}{'数值'.rjust(val_w)}")
    print("-" * (name_w + val_w))
    for name, value in rows:
        print(f"{name.ljust(name_w)}{value:>{val_w}.4f}")

print_kpi_table(kpi_rows)

# =========================
# 7) 绘图
# =========================
fig, axes = plt.subplots(3, 1, figsize=(10, 10), constrained_layout=True)

# 图1：水深沿程分布
axes[0].plot(x / 1000.0, y, lw=2, label="水深 y(x)")
axes[0].axhline(yn, color="tab:orange", ls="--", label=f"正常水深 yn={yn:.2f} m")
axes[0].set_xlabel("里程 x (km)")
axes[0].set_ylabel("水深 (m)")
axes[0].set_title("渐变流水深分布")
axes[0].grid(alpha=0.3)
axes[0].legend()

# 图2：河床线与水面线
axes[1].plot(x / 1000.0, z_bed, lw=2, label="河床线 z_bed")
axes[1].plot(x / 1000.0, z_ws, lw=2, label="水面线 z_ws")
axes[1].set_xlabel("里程 x (km)")
axes[1].set_ylabel("高程 (m)")
axes[1].set_title("沿程河床-水面线")
axes[1].grid(alpha=0.3)
axes[1].legend()

# 图3：富余水深安全校核
axes[2].plot(x / 1000.0, ukc, lw=2, label="UKC")
axes[2].axhline(ukc_required, color="tab:red", ls="--", label=f"要求值={ukc_required:.2f} m")
axes[2].fill_between(
    x / 1000.0, ukc, ukc_required,
    where=(ukc < ukc_required), alpha=0.25, color="red", label="不满足区段"
)
axes[2].set_xlabel("里程 x (km)")
axes[2].set_ylabel("富余水深 (m)")
axes[2].set_title("通航安全校核（考虑下沉量）")
axes[2].grid(alpha=0.3)
axes[2].legend()

plt.show()
```

### 1. 理论推导详述：明渠渐变流与通航安全的物理机制

上述仿真代码的实现依赖于严密的流体力学与航道工程理论框架。本小节将对代码中隐含的核心控制方程及经验模型进行全维度的理论推导，建立物理过程与数值逻辑的映射关系。

#### 1.1 恒定渐变流基本微分方程的推导
内河航道水面线的推算基础是恒定渐变流微分方程。在恒定流假设下，提取一段长度为 $dx$ 的微体作为控制体。基于实际流体的伯努利方程（Bernoulli Equation），沿程相邻两个断面（断面1与断面2）的能量守恒关系可表达为：
$$ Z_1 + h_1 + \frac{\alpha_1 v_1^2}{2g} = Z_2 + h_2 + \frac{\alpha_2 v_2^2}{2g} + h_w $$
式中：$Z$ 为河床底面高程；$h$ 为水深；$v$ 为断面平均流速；$\alpha$ 为动能修正系数；$h_w$ 为微段内的沿程水头损失。
对方程两侧沿流动方向 $x$ 求导，设定微段长度趋于无穷小，可得比能变化率：
$$ \frac{d}{dx}\left(Z + h + \frac{\alpha v^2}{2g}\right) = -J $$
其中 $J = h_w / dx$ 为水力坡度（沿程水头损失率）。根据河道地形几何关系，河床底坡 $S_0 = -dZ/dx$。若忽略流速分布的极端畸变，取动能修正系数 $\alpha \approx 1.0$，对流速水头项应用链式求导法则（考虑流量 $Q$ 为常数，流速 $v = Q/A$）：
$$ \frac{d}{dx}\left(\frac{v^2}{2g}\right) = \frac{1}{2g} \frac{d}{dx}\left(\frac{Q^2}{A^2}\right) = -\frac{Q^2}{g A^3} \frac{dA}{dx} $$
针对宽度为 $B$ 的明渠（如代码中的矩形断面航道），微小过水面积变化与水深变化的关系为 $dA = B dh$，代入上式得到：
$$ -\frac{Q^2 B}{g A^3} \frac{dh}{dx} = -\frac{v^2 B}{g A} \frac{dh}{dx} $$
引入水力学无量纲量——弗劳德数（Froude Number），其定义式为 $Fr = \frac{v}{\sqrt{g h_m}}$，其中 $h_m = A/B$ 为平均水深。代换后可知流速水头的导数项实则为 $-Fr^2 \frac{dh}{dx}$。
将所有项整合，得到水深沿程变化的常微分方程：
$$ S_0 - \frac{dh}{dx} - Fr^2 \frac{dh}{dx} = S_f $$
整理即可得出代码中 `gvf_ode` 函数所依据的核心理论模型：
$$ \frac{dh}{dx} = \frac{S_0 - S_f}{1 - Fr^2} $$
其中，摩阻坡降 $S_f$ 依托曼宁（Manning）公式推导计算，其力学本源来自于床面剪切应力 $\tau_0 = \gamma R S_f$ 与谢才公式 $v = C \sqrt{R S_f}$ 的结合。在曼宁体系中，谢才系数 $C = \frac{1}{n} R^{1/6}$，反推即得代码中的表达形式：$S_f = \frac{n^2 v^2}{R^{4/3}}$。

#### 1.2 船舶下沉量（Squat）的流体力学机理
仿真代码中的 `squat = squat_coeff * V ** 2 / (2.0 * g)` 虽然表现为简化的经验函数，其背后却有着深厚的水动力学机制。当船舶在限制性航道中航行时，船体排开水体迫使周围水流加速回流。由伯努利原理可知，流速急剧增加导致船底局部动水压力急剧下降，造成船体整体垂直下沉及纵倾，这一现象称为 Squat 效应。
依据 Schijf 的一维极值理论，限制性航道内的质量守恒方程约束了回流速度 $\Delta v$ 与水位下降 $\Delta h$ 的关系：
$$ v_s A_c = (v_s + \Delta v)(A_c - A_s - B_c \Delta h) $$
式中：$v_s$ 为船舶对水航速（在顺流/逆流模型中等效考虑自然流速 $V$）；$A_c$ 为航道过水面积；$A_s$ 为船舯浸水面积。
由于一维理论解析解存在复杂的超越方程求根问题，工程界通常采用无量纲经验回归模型。如 Tuck 公式与 Barrass 公式均指出，船舶下沉量与傅汝德数平方（表征动能与重力势能之比）呈显著正相关，这构成了代码中将下沉量设为流速水头 $\frac{V^2}{2g}$ 倍数（即 `squat_coeff`）的坚实理论基础。

#### 1.3 富余水深（UKC）的边界约束分析
通航水力耦合评估的最终落脚点在于富余水深（Under Keel Clearance）。静水吃水并不能真实反映船舶航行时的极限受限状态。动态 UKC 的理论边界方程为：
$$ UKC_{dynamic} = h_{local} - T_{static} - S_{q}(v, h) - \Sigma \delta_i $$
方程中：$h_{local}$ 为当地瞬时水深，由常微分方程数值求解得出；$T_{static}$ 为船舶静吃水；$S_{q}$ 为水动力下沉量；$\Sigma \delta_i$ 代表各种不确定性裕度总和，包含波浪引起的升沉扰动、测深误差、以及潮汐或径流的不确定性。代码中设定的 `ukc_required` 即为工程规范所要求的动态富余水深下限。

### 2. 核心算法的数值方法解析

为实现上述偏微与代数方程组的闭环求解，代码采用了具备高鲁棒性的现代数值算法。深入解析这些算法有助于理解计算程序的收敛边界与稳定性特征。

#### 2.1 常微分方程的龙格-库塔数值积分（RK45）
在渐变流水面线推算中，`scipy.integrate.solve_ivp` 调用了 RK45（Dormand-Prince 方法）算法。这是一种具有自适应步长控制的显式龙格-库塔求解器。水面线方程 $\frac{dy}{dx} = f(x, y)$ 是典型的初值问题（IVP）。
传统四阶龙格-库塔法（RK4）通过在每一积分步长 $h$ 内进行四次斜率采样来逼近真实函数形态：
$$ k_1 = f(x_i, y_i) $$
$$ k_2 = f(x_i + \frac{h}{2}, y_i + \frac{h}{2}k_1) $$
$$ k_3 = f(x_i + \frac{h}{2}, y_i + \frac{h}{2}k_2) $$
$$ k_4 = f(x_i + h, y_i + h k_3) $$
$$ y_{i+1} = y_i + \frac{h}{6}(k_1 + 2k_2 + 2k_3 + k_4) $$
RK45 在 RK4 的基础上，利用同一组采样点构造一个五阶和一个四阶的近似解，利用两者之差估算局部截断误差。若误差大于设定的相对容差（`rtol=1e-6`）或绝对容差（`atol=1e-8`），算法将自动缩小积分步长 $dx$。这一机制极大地保障了河底坡降剧烈变化或水深产生急剧震荡区段的计算精度。

#### 2.2 Brent非线性方程求根算法
在求解正常水深 $y_n$ 时，代码定义了残差函数 `manning_residual`，并运用 `brentq` 算法。由于截面水力半径 $R = \frac{by}{b+2y}$ 是水深 $y$ 的高度非线性有理函数，解析求解曼宁方程并不现实。
Brent 算法被誉为单变量求根的黄金标准，其算法内核融合了二分法（Bisection Method）、割线法（Secant Method）以及逆二次插值（Inverse Quadratic Interpolation）。在迭代初期，若函数呈现强烈非线性，算法采用二分法保证绝对收敛；当逼近零点附近且曲线较为平滑时，自动切换至逆二次插值，实现超线性收敛（收敛阶数约为 $1.618$）。该算法不仅避免了牛顿-拉夫逊法（Newton-Raphson）对函数导数的繁琐计算，同时杜绝了导数接近零时可能引发的迭代发散风险。

#### 2.3 临界流奇异性及其数值稳定化处理
在渐变流方程 $\frac{dy}{dx} = \frac{S_0 - S_f}{1 - Fr^2}$ 中，当流态由缓流向急流演变（或相反），即弗劳德数 $Fr \to 1$ 时，分母趋向于零，导致水面变率趋于无限大，产生数学上的奇异性（Singularity）。
为防止数值积分在此处崩溃，代码中特别引入了稳健性保护机制：
```python
if abs(den) < 1e-4:
    den = 1e-4 if den >= 0 else -1e-4
```
这种截断处理在计算水力学中称为“极值阻尼修正”。从物理学视角审视，纯粹的一维渐变流假设在临界水深附近本就失效，流线将产生极度弯曲，出现水跃（Hydraulic Jump）或水跌。因此，微小的数值截断不仅保障了代码的稳定执行，且在不引入二维浅水波方程的条件下，给出了工程可接受的一维近似包络线。

### 4. 拓展视野：多维水动力耦合技术与数字孪生前沿

随着现代内河航运向深水化、大型化、智能化演进，传统的单断面一维恒定流计算已难以涵盖全部工程诉求。当前航道水力学的学科前沿正向以下两个维度深度扩展：

**其一，由一维数值积分迈向三维计算流体力学（CFD）。** 
在桥区水域、急弯航段或浅滩交汇区，水流不再服从一维流线平行的假定。工程界广泛引入基于雷诺平均纳维-斯托克斯方程（RANS）的三维 CFD 模型，结合 $k-\epsilon$ 或 $k-\omega$ 湍流模型，对二次流（Secondary Flow）及漩涡脱落进行高保真重现。在计算船舶动态下沉量时，通过嵌套网格（Overset Mesh）与流体体积法（VOF）捕捉自由液面，能够精准计算船-水-底三者的瞬态耦合水动力响应，取代传统的静态经验系数评估。

**其二，数字孪生（Digital Twin）与物理信息神经网络（PINN）。** 
依托 AIS（船舶自动识别系统）、多波束测深及水文遥测矩阵，航道正加速构建全生命周期数字孪生底座。在实时求解圣维南偏微分方程组的过程中，物理信息神经网络（PINN）展现出强大的潜能。它将流体力学控制方程的残差嵌入神经网络的损失函数中，在保障力学机理物理正确性的前提下，实现水深场与流速场毫秒级的推演，为船舶动态限载调度与实时 UKC 预警提供了代际跨越式的技术支撑。

# -*- coding: utf-8 -*-
"""
教材：《内河航道与通航水力学》
章节：航道渐变流与通航安全耦合仿真
功能：基于 Manning 公式与渐变流方程计算沿程水深，进一步评估船舶富余水深（UKC）、
输出 KPI 结果表，并绘制水力与通航安全图。
章节关键词：渐变流、Manning 糙率、弗劳德数、富余水深、通航能力
"""

import numpy as np
from scipy.integrate import solve_ivp
from scipy.optimize import brentq
import matplotlib.pyplot as plt

# =========================
# 1) 关键参数（可按教学场景调整）
# =========================
g = 9.81                  # 重力加速度 (m/s^2)
L = 50_000.0              # 航段长度 (m)
b = 80.0                  # 航道等效底宽 (m)
S0 = 1.2e-4               # 河床坡降 (-)
n = 0.028                 # Manning 糙率
Q = 900.0                 # 设计流量 (m^3/s)
y_upstream = 7.0          # 上游控制断面水深 (m)

ship_draft = 3.2          # 船舶吃水 (m)
squat_coeff = 0.15        # 下沉量系数（经验）
ukc_required = 0.8        # 最小富余水深要求 (m)

nx = 400                  # 沿程离散点数

# =========================
# 2) 水力学基础函数
# =========================
def area(y):
    """矩形断面过水面积 A"""
    return b * y

def wetted_perimeter(y):
    """矩形断面湿周 P"""
    return b + 2.0 * y

def hydraulic_radius(y):
    """水力半径 R = A/P"""
    return area(y) / wetted_perimeter(y)

def velocity(y):
    """断面平均流速 V = Q/A"""
    return Q / area(y)

def friction_slope(y):
    """摩阻坡降 Sf（Manning）"""
    A = area(y)
    R = hydraulic_radius(y)
    return (n * Q / (A * R ** (2.0 / 3.0))) ** 2

def froude_number(y):
    """弗劳德数 Fr = V/sqrt(g*y)（矩形断面）"""
    V = velocity(y)
    return V / np.sqrt(g * y)

# =========================
# 3) 求正常水深 yn（用于对比）
# =========================
def manning_residual(y):
    A = area(y)
    R = hydraulic_radius(y)
    q_calc = (1.0 / n) * A * R ** (2.0 / 3.0) * np.sqrt(S0)
    return q_calc - Q

yn = brentq(manning_residual, 0.2, 30.0)

# =========================
# 4) 渐变流微分方程 dy/dx = (S0 - Sf)/(1 - Fr^2)
# =========================
def gvf_ode(x, y):
    yv = max(float(y[0]), 0.05)  # 防止数值迭代出现非物理负水深
    sf = friction_slope(yv)
    fr = froude_number(yv)

    den = 1.0 - fr ** 2
    # 避免分母过小导致数值发散
    if abs(den) < 1e-4:
        den = 1e-4 if den >= 0 else -1e-4

    dydx = (S0 - sf) / den
    return [dydx]

x_eval = np.linspace(0.0, L, nx)
sol = solve_ivp(
    gvf_ode, (0.0, L), [y_upstream],
    t_eval=x_eval, method="RK45",
    rtol=1e-6, atol=1e-8
)

if not sol.success:
    raise RuntimeError(f"数值积分失败：{sol.message}")

x = sol.t
y = np.maximum(sol.y[0], 0.05)

# 河床与水面线高程（以 x=0 断面河床高程为 0）
z_bed = -S0 * x
z_ws = z_bed + y

# =========================
# 5) 通航指标（UKC）计算
# =========================
V = velocity(y)
Fr = froude_number(y)
Sf = friction_slope(y)

# 船舶下沉量（简化经验模型）
squat = squat_coeff * V ** 2 / (2.0 * g)

# 富余水深 UKC = 实际水深 - 吃水 - 下沉量
ukc = y - ship_draft - squat
safe_mask = ukc >= ukc_required
safe_ratio = 100.0 * np.mean(safe_mask)

i_min_y = int(np.argmin(y))
i_min_ukc = int(np.argmin(ukc))

# =========================
# 6) KPI 结果表打印
# =========================
kpi_rows = [
    ("正常水深 yn (m)", yn),
    ("最小水深 y_min (m)", y[i_min_y]),
    ("最小水深位置 x (km)", x[i_min_y] / 1000.0),
    ("最大流速 V_max (m/s)", np.max(V)),
    ("最大弗劳德数 Fr_max (-)", np.max(Fr)),
    ("平均摩阻坡降 Sf_mean (-)", np.mean(Sf)),
    ("最小富余水深 UKC_min (m)", ukc[i_min_ukc]),
    ("UKC 满足率 (%)", safe_ratio),
]

def print_kpi_table(rows):
    name_w = max(len(r[0]) for r in rows) + 2
    val_w = 14
    print("\n=== KPI 结果表 ===")
    print(f"{'指标'.ljust(name_w)}{'数值'.rjust(val_w)}")
    print("-" * (name_w + val_w))
    for name, value in rows:
        print(f"{name.ljust(name_w)}{value:>{val_w}.4f}")

print_kpi_table(kpi_rows)

# =========================
# 7) 绘图
# =========================
fig, axes = plt.subplots(3, 1, figsize=(10, 10), constrained_layout=True)

# 图1：水深沿程分布
axes[0].plot(x / 1000.0, y, lw=2, label="水深 y(x)")
axes[0].axhline(yn, color="tab:orange", ls="--", label=f"正常水深 yn={yn:.2f} m")
axes[0].set_xlabel("里程 x (km)")
axes[0].set_ylabel("水深 (m)")
axes[0].set_title("渐变流水深分布")
axes[0].grid(alpha=0.3)
axes[0].legend()

# 图2：河床线与水面线
axes[1].plot(x / 1000.0, z_bed, lw=2, label="河床线 z_bed")
axes[1].plot(x / 1000.0, z_ws, lw=2, label="水面线 z_ws")
axes[1].set_xlabel("里程 x (km)")
axes[1].set_ylabel("高程 (m)")
axes[1].set_title("沿程河床-水面线")
axes[1].grid(alpha=0.3)
axes[1].legend()

# 图3：富余水深安全校核
axes[2].plot(x / 1000.0, ukc, lw=2, label="UKC")
axes[2].axhline(ukc_required, color="tab:red", ls="--", label=f"要求值={ukc_required:.2f} m")
axes[2].fill_between(
    x / 1000.0, ukc, ukc_required,
    where=(ukc < ukc_required), alpha=0.25, color="red", label="不满足区段"
)
axes[2].set_xlabel("里程 x (km)")
axes[2].set_ylabel("富余水深 (m)")
axes[2].set_title("通航安全校核（考虑下沉量）")
axes[2].grid(alpha=0.3)
axes[2].legend()

plt.show()
